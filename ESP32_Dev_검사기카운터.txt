/* ============================================================
   ESP32 DevKitC V4 - 2채널 SHOT 카운터 v2.0
   ============================================================
   DEVICE : ESP_S1
   채널   : S1=GPIO34 (릴레이), S2=GPIO35 (광센서)
   WiFi   : TAEJIN / taejin-A / Android0031 (자동 선택)
   ============================================================ */

#include <Arduino.h>
#include <WiFi.h>
#include <WiFiMulti.h>
#include <HTTPClient.h>
#include <WiFiClientSecure.h>
#include <ArduinoJson.h>
#include "esp_mac.h"

const char* DEVICE_ID = "ESP_S1";

// 채널 설정 (핀, machine_code)
const int   NUM_CH = 2;
const int   CH_PIN[NUM_CH]  = { 34, 35 };
const char* CH_CODE[NUM_CH] = { "S1", "S2" };

WiFiMulti wifiMulti;

const char* SUPABASE_URL = "https://omngtyewdaqpphnzeate.supabase.co/rest/v1/shot_count";
const char* SUPABASE_KEY = "sb_publishable_9j2YkkL-7ul1TrhH-NjVdQ_vWDG2-1D";

const unsigned long DEBOUNCE_MS      = 50;
const unsigned long WIFI_CHECK_MS    = 10000;
const unsigned long WIFI_REPORT_MS   = 60000;
const unsigned long DIAG_PERIOD_MS   = 5000;
const unsigned long POST_INTERVAL_MS = 50;

#define QUEUE_SIZE 60

// 큐 (채널 정보 포함)
int           q_ch[QUEUE_SIZE];
unsigned long q_count[QUEUE_SIZE];
unsigned long q_interval[QUEUE_SIZE];
int q_head    = 0;
int q_tail    = 0;
int q_dropped = 0;

byte  mac_addr[6];
char  mac_str[18];
char  ip_str[20];
String currentSSID = "";

// 채널별 상태
unsigned long shot_count[NUM_CH]   = {0};
unsigned long last_shot_ms[NUM_CH] = {0};
int           last_state[NUM_CH]   = {HIGH, HIGH};

unsigned long last_wifi_check  = 0;
unsigned long last_wifi_report = 0;
unsigned long last_diag        = 0;
unsigned long last_post_try    = 0;

int queueSize(){
  return (q_head - q_tail + QUEUE_SIZE) % QUEUE_SIZE;
}

bool enqueueShot(int ch, unsigned long cnt, unsigned long interval){
  int next = (q_head + 1) % QUEUE_SIZE;
  if(next == q_tail){ q_dropped++; return false; }
  q_ch[q_head]       = ch;
  q_count[q_head]    = cnt;
  q_interval[q_head] = interval;
  q_head = next;
  return true;
}

void updateIPInfo(){
  IPAddress ip = WiFi.localIP();
  snprintf(ip_str, sizeof(ip_str), "%d.%d.%d.%d", ip[0],ip[1],ip[2],ip[3]);
  currentSSID = WiFi.SSID();
}

void setupWiFi(){
  esp_read_mac(mac_addr, ESP_MAC_WIFI_STA);
  snprintf(mac_str, sizeof(mac_str), "%02X:%02X:%02X:%02X:%02X:%02X",
    mac_addr[0],mac_addr[1],mac_addr[2],mac_addr[3],mac_addr[4],mac_addr[5]);
  Serial.print("MAC: "); Serial.println(mac_str);

  WiFi.mode(WIFI_STA);
  wifiMulti.addAP("TAEJIN",      "taejinpress");
  wifiMulti.addAP("taejin-A",    "taejinpress");
  wifiMulti.addAP("Android0031", "");

  Serial.println("WiFi 검색 중 (TAEJIN / taejin-A / Android0031)...");

  unsigned long t0 = millis();
  while(wifiMulti.run() != WL_CONNECTED && millis()-t0 < 30000){
    Serial.print(".");
    delay(500);
  }
  Serial.println();

  if(WiFi.status() == WL_CONNECTED){
    updateIPInfo();
    Serial.println("WiFi 연결 성공!");
    Serial.print("   SSID : "); Serial.println(currentSSID);
    Serial.print("   IP   : "); Serial.println(ip_str);
    Serial.print("   RSSI : "); Serial.print(WiFi.RSSI()); Serial.println(" dBm");
  } else {
    Serial.println("WiFi 연결 실패");
    strcpy(ip_str, "0.0.0.0");
    currentSSID = "없음";
  }
}

void checkWiFi(){
  unsigned long now = millis();
  if(now - last_wifi_check < WIFI_CHECK_MS) return;
  last_wifi_check = now;

  wifiMulti.run();

  if(WiFi.status() != WL_CONNECTED){
    strcpy(ip_str, "0.0.0.0");
    currentSSID = "재연결중";
    return;
  }
  updateIPInfo();

  if(now - last_wifi_report >= WIFI_REPORT_MS){
    last_wifi_report = now;
    Serial.printf("WiFi OK | SSID=%s IP=%s RSSI=%ddBm Q=%d S1=%lu S2=%lu\n",
      currentSSID.c_str(), ip_str, WiFi.RSSI(), queueSize(),
      shot_count[0], shot_count[1]);
  }
}

void testSupabase(){
  if(WiFi.status() != WL_CONNECTED){ Serial.println("[TEST] WiFi 없음"); return; }
  Serial.println("[TEST] Supabase 연결 확인 중...");

  WiFiClientSecure client;
  client.setInsecure();
  HTTPClient https;
  String url = String(SUPABASE_URL) + "?limit=1";

  if(!https.begin(client, url)){ Serial.println("[TEST] HTTPS 실패"); return; }
  https.addHeader("apikey", SUPABASE_KEY);
  https.addHeader("Authorization", String("Bearer ") + SUPABASE_KEY);
  https.setTimeout(10000);

  int code = https.GET();
  https.end();

  if(code >= 200 && code < 300) Serial.println("[TEST] Supabase 접속 OK");
  else if(code == -1)           Serial.println("[TEST] 인터넷 불가");
  else                          Serial.printf("[TEST] HTTP=%d\n", code);
}

bool processOneQueueItem(){
  if(q_head == q_tail) return false;
  if(WiFi.status() != WL_CONNECTED) return false;

  int           ch       = q_ch[q_tail];
  unsigned long cnt      = q_count[q_tail];
  unsigned long interval = q_interval[q_tail];
  int saved_tail = q_tail;
  q_tail = (q_tail + 1) % QUEUE_SIZE;

  updateIPInfo();

  StaticJsonDocument<256> doc;
  doc["device_id"]        = DEVICE_ID;
  doc["machine_code"]     = CH_CODE[ch];
  doc["shot_count"]       = cnt;
  doc["shot_interval_ms"] = interval;
  doc["esp32_mac"]        = mac_str;
  doc["esp32_ip"]         = ip_str;

  String body;
  serializeJson(doc, body);

  WiFiClientSecure client;
  client.setInsecure();
  HTTPClient https;

  if(!https.begin(client, SUPABASE_URL)){
    q_tail = saved_tail;
    return false;
  }

  https.addHeader("Content-Type",  "application/json");
  https.addHeader("apikey",        SUPABASE_KEY);
  https.addHeader("Authorization", String("Bearer ") + SUPABASE_KEY);
  https.addHeader("Prefer",        "return=minimal");

  int code = https.POST(body);
  https.end();

  bool ok = (code >= 200 && code < 300);
  Serial.printf("POST %s #%lu Q=%d -> %d%s\n",
    CH_CODE[ch], cnt, queueSize(), code, ok ? " OK" : " FAIL");

  if(!ok) q_tail = saved_tail;
  return ok;
}

void checkSensors(){
  unsigned long now = millis();
  for(int ch = 0; ch < NUM_CH; ch++){
    int raw = digitalRead(CH_PIN[ch]);
    if(last_state[ch] == HIGH && raw == LOW){
      if(now - last_shot_ms[ch] >= DEBOUNCE_MS){
        shot_count[ch]++;
        unsigned long interval = (last_shot_ms[ch] == 0) ? 0 : (now - last_shot_ms[ch]);
        last_shot_ms[ch] = now;
        bool queued = enqueueShot(ch, shot_count[ch], interval);
        Serial.printf("SHOT %s #%lu interval=%lums Q=%d%s\n",
          CH_CODE[ch], shot_count[ch], interval, queueSize(), queued ? "" : " 누락");
      }
    }
    last_state[ch] = raw;
  }
}

void printDiag(){
  Serial.printf("[DIAG] WiFi=%s IP=%s | ",
    WiFi.status()==WL_CONNECTED ? currentSSID.c_str() : "OFF", ip_str);
  for(int ch = 0; ch < NUM_CH; ch++){
    int lv = digitalRead(CH_PIN[ch]);
    Serial.printf("%s(G%d)=%s cnt=%lu | ",
      CH_CODE[ch], CH_PIN[ch],
      lv==LOW ? "LOW" : "HIGH", shot_count[ch]);
  }
  Serial.printf("Q=%d 누락=%d\n", queueSize(), q_dropped);
}

void setup(){
  Serial.begin(115200);
  delay(800);
  Serial.println();
  Serial.println("=========================================");
  Serial.printf ("ESP_S1 2채널 SHOT Counter v2.0  [%s]\n", DEVICE_ID);
  Serial.println("=========================================");
  for(int ch = 0; ch < NUM_CH; ch++){
    pinMode(CH_PIN[ch], INPUT);
    last_state[ch] = digitalRead(CH_PIN[ch]);
    Serial.printf("CH%d: GPIO%d -> %s\n", ch+1, CH_PIN[ch], CH_CODE[ch]);
  }

  setupWiFi();
  testSupabase();

  Serial.println("\nReady. 신호 감지 대기 중...\n");
  last_diag = millis();
}

void loop(){
  checkWiFi();
  checkSensors();

  unsigned long now = millis();
  if(now - last_post_try >= POST_INTERVAL_MS){
    last_post_try = now;
    processOneQueueItem();
  }

  if(now - last_diag >= DIAG_PERIOD_MS){
    last_diag = now;
    printDiag();
  }
}
